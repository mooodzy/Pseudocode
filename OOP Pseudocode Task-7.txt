Task 308: Insurance Claims Processing 

CLASS Claim

    ATTRIBUTES
        claimNumber : String
        dateFiled : String
        status : String
        documentationFolder : DocumentationFolder

    CONSTRUCTOR Claim(claimNumber : String, dateFiled : String, status : String)

        SET this.claimNumber = claimNumber
        SET this.dateFiled = dateFiled
        SET this.status = status
        SET documentationFolder = NEW DocumentationFolder()

    METHODS

        calculatePayout() : Double

            RETURN 0

END CLASS


CLASS VehicleClaim INHERITS Claim

    ATTRIBUTES
        repairEstimate : Double
        excess : Double

    CONSTRUCTOR VehicleClaim(claimNumber : String, dateFiled : String, status : String, repairEstimate : Double, excess : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.repairEstimate = repairEstimate
        SET this.excess = excess

    OVERRIDE METHOD calculatePayout() : Double

        RETURN repairEstimate - excess

END CLASS


CLASS MedicalClaim INHERITS Claim

    ATTRIBUTES
        treatmentCost : Double
        medicalCoverageLimit : Double

    CONSTRUCTOR MedicalClaim(claimNumber : String, dateFiled : String, status : String, treatmentCost : Double, medicalCoverageLimit : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.treatmentCost = treatmentCost
        SET this.medicalCoverageLimit = medicalCoverageLimit

    OVERRIDE METHOD calculatePayout() : Double

        IF treatmentCost > medicalCoverageLimit THEN

            RETURN medicalCoverageLimit

        ELSE

            RETURN treatmentCost

        END IF

END CLASS


CLASS PropertyClaim INHERITS Claim

    ATTRIBUTES
        damageValue : Double
        maintenanceReduction : Double

    CONSTRUCTOR PropertyClaim(claimNumber : String, dateFiled : String, status : String, damageValue : Double, maintenanceReduction : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.damageValue = damageValue
        SET this.maintenanceReduction = maintenanceReduction

    OVERRIDE METHOD calculatePayout() : Double

        RETURN damageValue - (damageValue * maintenanceReduction / 100)

END CLASS


CLASS DocumentationFolder

    ATTRIBUTES
        documents : List(String)

    CONSTRUCTOR DocumentationFolder()

        CREATE empty documents list

END CLASS


CLASS Policy

    ATTRIBUTES
        policyNumber : String
        claims : List(Claim)

    CONSTRUCTOR Policy(policyNumber : String)

        SET this.policyNumber = policyNumber

        CREATE empty claims list

    METHODS

        addClaim(claim : Claim)

            ADD claim TO claims

        processClaims()

            SET i = 0

            WHILE i < size of claims

                PRINT claims[i].claimNumber
                PRINT claims[i].calculatePayout()

                SET i = i + 1

            END WHILE

END CLASS


START PROGRAM

DECLARE policy : Policy

SET policy = NEW Policy("POL001")


DECLARE vehicle : VehicleClaim

SET vehicle = NEW VehicleClaim("VC001","10/05/2026","Pending",5000,500)


DECLARE medical : MedicalClaim

SET medical = NEW MedicalClaim("MC001","11/05/2026","Pending",8000,6000)


DECLARE property : PropertyClaim

SET property = NEW PropertyClaim("PC001","12/05/2026","Pending",12000,10)


CALL policy.addClaim(vehicle)

CALL policy.addClaim(medical)

CALL policy.addClaim(property)


PRINT "Vehicle Claim Payout = "

PRINT vehicle.calculatePayout()


PRINT "Medical Claim Payout = "

PRINT medical.calculatePayout()


PRINT "Property Claim Payout = "

PRINT property.calculatePayout()


CALL policy.processClaims()

END PROGRAM



Task 309: University Library Loans (Medium)


CLASS LoanItem

    ATTRIBUTES
        title : String
        barcode : String
        loanDuration : Integer
        conditionReport : ConditionReport


    CONSTRUCTOR LoanItem(title : String, barcode : String, loanDuration : Integer)

        SET this.title = title
        SET this.barcode = barcode
        SET this.loanDuration = loanDuration

        SET conditionReport = NEW ConditionReport()


    METHODS

        calculateLateFee(daysOverdue : Integer) : Double

            RETURN 0


        renewLoan() : Boolean

            RETURN TRUE

END CLASS



CLASS StandardBook INHERITS LoanItem

    ATTRIBUTES
        dailyLateFee : Double


    CONSTRUCTOR StandardBook(title : String, barcode : String, loanDuration : Integer)

        CALL SUPER(title, barcode, loanDuration)

        SET dailyLateFee = 0.50


    OVERRIDE METHOD calculateLateFee(daysOverdue : Integer) : Double

        RETURN daysOverdue * dailyLateFee

END CLASS



CLASS ReferenceTextbook INHERITS LoanItem

    ATTRIBUTES
        dailyLateFee : Double


    CONSTRUCTOR ReferenceTextbook(title : String, barcode : String, loanDuration : Integer)

        CALL SUPER(title, barcode, loanDuration)

        SET dailyLateFee = 2.00


    OVERRIDE METHOD calculateLateFee(daysOverdue : Integer) : Double

        RETURN daysOverdue * dailyLateFee


    OVERRIDE METHOD renewLoan() : Boolean

        RETURN FALSE

END CLASS



CLASS JournalIssue INHERITS ReferenceTextbook


    CONSTRUCTOR JournalIssue(title : String, barcode : String, loanDuration : Integer)

        CALL SUPER(title, barcode, loanDuration)


    OVERRIDE METHOD renewLoan() : Boolean

        RETURN FALSE


    METHOD canRenewAfterOverdue() : Boolean

        RETURN FALSE

END CLASS



CLASS ConditionReport

    ATTRIBUTES
        damageNotes : String


    CONSTRUCTOR ConditionReport()

        SET damageNotes = "No damage"


    METHODS

        addDamageNote(note : String)

            SET damageNotes = note

END CLASS



CLASS StudentAccount

    ATTRIBUTES
        studentName : String
        loanItems : List<LoanItem>


    CONSTRUCTOR StudentAccount(studentName : String)

        SET this.studentName = studentName

        CREATE empty loanItems list


    METHODS


        addLoanItem(item : LoanItem)

            ADD item TO loanItems



        calculateAllLateFees(daysOverdue : Integer)

            SET i = 0

            WHILE i < size of loanItems

                PRINT loanItems[i].title

                PRINT loanItems[i].calculateLateFee(daysOverdue)

                SET i = i + 1

            END WHILE


END CLASS



START PROGRAM


DECLARE account : StudentAccount

SET account = NEW StudentAccount("Mohammed")



DECLARE book : StandardBook

DECLARE textbook : ReferenceTextbook

CALL account.addLoanItem(textbook)

Task 310: Theme Park Ride Queues



CLASS Ride

    ATTRIBUTES
        name : String
        location : String
        safetyLog : SafetyInspectionLog


    CONSTRUCTOR Ride(name : String, location : String)

        SET this.name = name
        SET this.location = location

        SET safetyLog = NEW SafetyInspectionLog()


    METHODS

        calculateWaitTime(numberOfPeople : Integer) : Double

            RETURN 0

END CLASS



CLASS RollerCoaster INHERITS Ride

    ATTRIBUTES
        peoplePerTrainCar : Integer
        rideCycleTime : Double


    CONSTRUCTOR RollerCoaster(name : String, location : String,
                              peoplePerTrainCar : Integer, rideCycleTime : Double)

        CALL SUPER(name, location)

        SET this.peoplePerTrainCar = peoplePerTrainCar
        SET this.rideCycleTime = rideCycleTime


    OVERRIDE METHOD calculateWaitTime(numberOfPeople : Integer) : Double

        SET numberOfCars = numberOfPeople / peoplePerTrainCar

        RETURN numberOfCars * rideCycleTime

END CLASS



CLASS WaterRide INHERITS Ride

    ATTRIBUTES
        peoplePerBoat : Integer
        rideCycleTime : Double
        loadingTime : Double


    CONSTRUCTOR WaterRide(name : String, location : String,
                          peoplePerBoat : Integer,
                          rideCycleTime : Double,
                          loadingTime : Double)

        CALL SUPER(name, location)

        SET this.peoplePerBoat = peoplePerBoat
        SET this.rideCycleTime = rideCycleTime
        SET this.loadingTime = loadingTime


    OVERRIDE METHOD calculateWaitTime(numberOfPeople : Integer) : Double

        SET numberOfBoats = numberOfPeople / peoplePerBoat

        RETURN numberOfBoats * (rideCycleTime + loadingTime)

END CLASS



CLASS VIPAccessRide INHERITS RollerCoaster


    CONSTRUCTOR VIPAccessRide(name : String, location : String,
                              peoplePerTrainCar : Integer,
                              rideCycleTime : Double)

        CALL SUPER(name, location, peoplePerTrainCar, rideCycleTime)



    METHOD calculateVIPWaitTime(numberOfPeople : Integer,
                                hasVIPPass : Boolean) : Double


        IF hasVIPPass = TRUE THEN

            RETURN 0

        ELSE

            RETURN calculateWaitTime(numberOfPeople)

        END IF


END CLASS



CLASS SafetyInspectionLog

    ATTRIBUTES
        inspectionDate : String
        inspectionOutcome : String


    CONSTRUCTOR SafetyInspectionLog()

        SET inspectionDate = "Not Checked"

        SET inspectionOutcome = "Unknown"



    METHODS

        updateInspection(date : String, outcome : String)

            SET inspectionDate = date

            SET inspectionOutcome = outcome


END CLASS



CLASS Park

    ATTRIBUTES
        rides : List<Ride>


    CONSTRUCTOR Park()

        CREATE empty rides list



    METHODS


        addRide(ride : Ride)

            ADD ride TO rides



        displayWaitTimes(numberOfPeople : Integer)

            SET i = 0


            WHILE i < size of rides


                PRINT rides[i].name

                PRINT rides[i].calculateWaitTime(numberOfPeople)


                SET i = i + 1


            END WHILE


END CLASS



START PROGRAM


DECLARE park : Park

SET park = NEW Park()



DECLARE roller : RollerCoaster

SET roller = NEW RollerCoaster("Thunder Mountain",
                               "Zone A",
                               20,
                               5)



DECLARE water : WaterRide

SET water = NEW WaterRide("River Adventure",
                          "Zone B",
                          10,
                          6,
                          2)



DECLARE vipRide : VIPAccessRide

SET vipRide = NEW VIPAccessRide("VIP Rocket",
                                "Zone C",
                                20,
                                5)



CALL park.addRide(roller)

CALL park.addRide(water)

CALL park.addRide(vipRide)



PRINT "Normal Guest Wait Times"


CALL park.displayWaitTimes(100)



PRINT "VIP Guest Wait Time"


PRINT vipRide.calculateVIPWaitTime(100, TRUE)


END PROGRAM
