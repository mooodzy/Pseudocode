Task 308: Insurance Claims Processing 

CLASS Claim

    ATTRIBUTES
        claimNumber : String
        dateFiled : String
        status : String
        documentationFolder : DocumentationFolder

    CONSTRUCTOR Claim(claimNumber : String, dateFiled : String, status : String)

        SET this.claimNumber = claimNumber
        SET this.dateFiled = dateFiled
        SET this.status = status
        SET documentationFolder = NEW DocumentationFolder()

    METHODS

        calculatePayout() : Double

            RETURN 0

END CLASS


CLASS VehicleClaim INHERITS Claim

    ATTRIBUTES
        repairEstimate : Double
        excess : Double

    CONSTRUCTOR VehicleClaim(claimNumber : String, dateFiled : String, status : String, repairEstimate : Double, excess : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.repairEstimate = repairEstimate
        SET this.excess = excess

    OVERRIDE METHOD calculatePayout() : Double

        RETURN repairEstimate - excess

END CLASS


CLASS MedicalClaim INHERITS Claim

    ATTRIBUTES
        treatmentCost : Double
        medicalCoverageLimit : Double

    CONSTRUCTOR MedicalClaim(claimNumber : String, dateFiled : String, status : String, treatmentCost : Double, medicalCoverageLimit : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.treatmentCost = treatmentCost
        SET this.medicalCoverageLimit = medicalCoverageLimit

    OVERRIDE METHOD calculatePayout() : Double

        IF treatmentCost > medicalCoverageLimit THEN

            RETURN medicalCoverageLimit

        ELSE

            RETURN treatmentCost

        END IF

END CLASS


CLASS PropertyClaim INHERITS Claim

    ATTRIBUTES
        damageValue : Double
        maintenanceReduction : Double

    CONSTRUCTOR PropertyClaim(claimNumber : String, dateFiled : String, status : String, damageValue : Double, maintenanceReduction : Double)

        CALL SUPER(claimNumber, dateFiled, status)

        SET this.damageValue = damageValue
        SET this.maintenanceReduction = maintenanceReduction

    OVERRIDE METHOD calculatePayout() : Double

        RETURN damageValue - (damageValue * maintenanceReduction / 100)

END CLASS


CLASS DocumentationFolder

    ATTRIBUTES
        documents : List(String)

    CONSTRUCTOR DocumentationFolder()

        CREATE empty documents list

END CLASS


CLASS Policy

    ATTRIBUTES
        policyNumber : String
        claims : List(Claim)

    CONSTRUCTOR Policy(policyNumber : String)

        SET this.policyNumber = policyNumber

        CREATE empty claims list

    METHODS

        addClaim(claim : Claim)

            ADD claim TO claims

        processClaims()

            SET i = 0

            WHILE i < size of claims

                PRINT claims[i].claimNumber
                PRINT claims[i].calculatePayout()

                SET i = i + 1

            END WHILE

END CLASS


START PROGRAM

DECLARE policy : Policy

SET policy = NEW Policy("POL001")


DECLARE vehicle : VehicleClaim

SET vehicle = NEW VehicleClaim("VC001","10/05/2026","Pending",5000,500)

SET medical = NEW MedicalClaim("MC001","11/05/2026","Pending",8000,6000)


DECLARE property : PropertyClaim

SET property = NEW PropertyClaim("PC001","12/05/2026","Pending",12000,10)


CALL policy.addClaim(vehicle)

CALL policy.addClaim(medical)

CALL policy.addClaim(property)


PRINT "Vehicle Claim Payout = "

PRINT vehicle.calculatePayout()


PRINT "Medical Claim Payout = "

CLASS Ride

    ATTRIBUTES
        name : String
        location : String
        safetyLog : SafetyInspectionLog


    CONSTRUCTOR Ride(name : String, location : String)

        SET this.name = name
        SET this.location = location

        SET safetyLog = NEW SafetyInspectionLog()


    METHODS

        calculateWaitTime(numberOfPeople : Integer) 
            RETURN 0

END CLASS



CLASS RollerCoaster INHERITS Ride

    ATTRIBUTES
        peoplePerTrainCar : Integer
        rideCycleTime : Double


    CONSTRUCTOR RollerCoaster(name : String, location : String,peoplePerTrainCar : Integer, rideCycleTime : Double)

        CALL SUPER(name, location)

        SET this.peoplePerTrainCar = peoplePerTrainCar
        SET this.rideCycleTime = rideCycleTime


    OVERRIDE METHOD calculateWaitTime(numberOfPeople : Integer) : Double

        SET numberOfCars = numberOfPeople / peoplePerTrainCar

        RETURN numberOfCars * rideCycleTime

END CLASS



CLASS WaterRide INHERITS Ride

    ATTRIBUTES
        peoplePerBoat : Integer
        rideCycleTime : Double
        loadingTime : Double


    CONSTRUCTOR WaterRide(name : String, location : String,peoplePerBoat : Integer,rideCycleTime : Double,loadingTime : Double)

        CALL SUPER(name, location)

        SET this.peoplePerBoat = peoplePerBoat
        SET this.rideCycleTime = rideCycleTime
        SET this.loadingTime = loadingTime


    OVERRIDE METHOD calculateWaitTime(numberOfPeople : Integer) : Double

        SET numberOfBoats = numberOfPeople / peoplePerBoat

        RETURN numberOfBoats * (rideCycleTime + loadingTime)

END CLASS



CLASS VIPAccessRide INHERITS RollerCoaster


    CONSTRUCTOR VIPAccessRide(name : String, location : String,
                              peoplePerTrainCar : Integer,
                              rideCycleTime : Double)

        CALL SUPER(name, location, peoplePerTrainCar, rideCycleTime)



    METHOD calculateVIPWaitTime(numberOfPeople : Integer,hasVIPPass : Boolean) : Double


        IF hasVIPPass = TRUE THEN

            RETURN 0

        ELSE

            RETURN calculateWaitTime(numberOfPeople)

        END IF


END CLASS

    METHODS

CLASS Invoice (ABSTRACT)

    ATTRIBUTES
        invoiceNumber : String
        dateIssued : String
        clientName : String
        paymentTerms : PaymentTerms


    CONSTRUCTOR Invoice(invoiceNumber : String,dateIssued : String,clientName : String,paymentTerms : PaymentTerms)

        SET this.invoiceNumber = invoiceNumber

        SET this.dateIssued = dateIssued

        SET this.clientName = clientName

        SET this.paymentTerms = paymentTerms



    METHOD getInvoiceNumber() : String

        RETURN invoiceNumber



    ABSTRACT METHOD calculateTotalAmount() : Double


END CLASS



CLASS FixedPriceInvoice INHERITS Invoice

    ATTRIBUTES
        projectFee : Double


    CONSTRUCTOR FixedPriceInvoice(invoiceNumber : String,dateIssued : String,clientName : String,paymentTerms : PaymentTerms,projectFee : Double)

        CALL SUPER(invoiceNumber, dateIssued, clientName, paymentTerms)

        SET this.projectFee = projectFee



    OVERRIDE METHOD calculateTotalAmount() : Double

        RETURN projectFee


END CLASS



CLASS HourEntry

    ATTRIBUTES
        hours : Double
        hourlyRate : Double
        approved : Boolean


    CONSTRUCTOR HourEntry(hours : Double,hourlyRate : Double, approved : Boolean)

        SET this.hours = hours

        SET this.hourlyRate = hourlyRate

        SET this.approved = approved



END CLASS



CLASS HourlyInvoice INHERITS Invoice

    ATTRIBUTES
        hourEntries : List(HourEntry)


    CONSTRUCTOR HourlyInvoice(invoiceNumber : String, dateIssued : String, clientName : String,paymentTerms : PaymentTerms)

        CALL SUPER(invoiceNumber, dateIssued, clientName, paymentTerms)

        CREATE empty hourEntries list



    METHOD addHourEntry(entry : HourEntry)

        ADD entry TO hourEntries



    OVERRIDE METHOD calculateTotalAmount() : Double

        DECLARE total : Double

        SET total = 0


        FOR EACH entry IN hourEntries

            IF entry.approved = TRUE THEN

                SET total = total + (entry.hours * entry.hourlyRate)

            END IF


        END FOR


        RETURN total


END CLASS



CLASS RetainerInvoice INHERITS FixedPriceInvoice

    ATTRIBUTES
        includedHours : Double
        extraHours : Double
        extraHourlyRate : Double


    CONSTRUCTOR RetainerInvoice(invoiceNumber : String,dateIssued : String,clientName : String,paymentTerms : PaymentTerms,projectFee : Double,includedHours : Double,extraHours : Double,extraHourlyRate : Double)

        CALL SUPER(invoiceNumber,dateIssued,clientName,paymentTerms,projectFee)


        SET this.includedHours = includedHours

        SET this.extraHours = extraHours

        SET this.extraHourlyRate = extraHourlyRate



    OVERRIDE METHOD calculateTotalAmount() : Double

        DECLARE total : Double

        SET total = SUPER.calculateTotalAmount()


        IF extraHours > includedHours THEN

            SET total = total + ((extraHours - includedHours) * extraHourlyRate)

        END IF


        RETURN total


END CLASS



CLASS PaymentTerms

    ATTRIBUTES
        dueDate : String
        lateInterestRate : Double


    CONSTRUCTOR PaymentTerms(dueDate : String,lateInterestRate : Double)

        SET this.dueDate = dueDate

        SET this.lateInterestRate = lateInterestRate


END CLASS



CLASS FreelancerAccount

    ATTRIBUTES
        invoices : List<Invoice>


    CONSTRUCTOR FreelancerAccount()

        CREATE empty invoices list



    METHOD addInvoice(invoice : Invoice)

        ADD invoice TO invoices



    METHOD displayInvoices()

        FOR EACH invoice IN invoices

            PRINT "Invoice Number: " + invoice.getInvoiceNumber()

            PRINT "Total Amount: " + invoice.calculateTotalAmount()


        END FOR


END CLASS

CLASS Server (ABSTRACT)

    ATTRIBUTES
        serialNumber : String
        uptime : Double
        hardware : HardwareSpecification


    CONSTRUCTOR Server(serialNumber : String,
                       uptime : Double,
                       hardware : HardwareSpecification)

        SET this.serialNumber = serialNumber

        SET this.uptime = uptime

        SET this.hardware = hardware



    METHOD getSerialNumber() : String

        RETURN serialNumber



    ABSTRACT METHOD calculateLoadPercentage() : Double


END CLASS



CLASS WebServer INHERITS Server

    ATTRIBUTES
        activeConnections : Integer
        maxConnections : Integer


    CONSTRUCTOR WebServer(serialNumber : String,
                          uptime : Double,
                          hardware : HardwareSpecification,
                          activeConnections : Integer,
                          maxConnections : Integer)

        CALL SUPER(serialNumber, uptime, hardware)

        SET this.activeConnections = activeConnections

        SET this.maxConnections = maxConnections



    OVERRIDE METHOD calculateLoadPercentage() : Double

        RETURN (activeConnections / maxConnections) * 100


END CLASS



CLASS DatabaseServer INHERITS Server

    ATTRIBUTES
        currentQueries : Integer
        maxQueriesPerSecond : Integer


    CONSTRUCTOR DatabaseServer(serialNumber : String,
                               uptime : Double,
                               hardware : HardwareSpecification,
                               currentQueries : Integer,
                               maxQueriesPerSecond : Integer)

        CALL SUPER(serialNumber, uptime, hardware)

        SET this.currentQueries = currentQueries

        SET this.maxQueriesPerSecond = maxQueriesPerSecond



    METHOD acceptConnection() : Boolean

        IF currentQueries >= maxQueriesPerSecond THEN

            PRINT "Server is at full capacity"

            RETURN FALSE

        ELSE

            SET currentQueries = currentQueries + 1

            RETURN TRUE

        END IF



    OVERRIDE METHOD calculateLoadPercentage() : Double

        RETURN (currentQueries / maxQueriesPerSecond) * 100


END CLASS



CLASS CacheServer INHERITS DatabaseServer

    ATTRIBUTES
        cacheHitRatio : Double


    CONSTRUCTOR CacheServer(serialNumber : String,
                            uptime : Double,
                            hardware : HardwareSpecification,
                            currentQueries : Integer,
                            maxQueriesPerSecond : Integer,
                            cacheHitRatio : Double)

        CALL SUPER(serialNumber,
                   uptime,
                   hardware,
                   currentQueries,
                   maxQueriesPerSecond)


        SET this.cacheHitRatio = cacheHitRatio



    OVERRIDE METHOD calculateLoadPercentage() : Double

        DECLARE load : Double


        SET load = SUPER.calculateLoadPercentage()


        SET load = load * (1 - cacheHitRatio)


        RETURN load


END CLASS



CLASS HardwareSpecification

    ATTRIBUTES
        cpu : String

        storage : Double


    CONSTRUCTOR HardwareSpecification(cpu : String,
                                      ram : Double,
                                      storage : Double)

        SET this.cpu = cpu

        SET this.storage = storage


END CLASS



CLASS Rack

    ATTRIBUTES
        servers : List<Server>


    CONSTRUCTOR Rack()

    METHOD addServer(server : Server)

    METHOD displayServers()

        FOR EACH server IN servers

            PRINT "Server Serial: " + server.getSerialNumber()

            PRINT "Load: " + server.calculateLoadPercentage()


        END FOR


END CLASS
