Task 293: Shape Drawing App :



CLASS Shape
    ATTRIBUTES
        color : String
        x : Integer
        y : Integer

    CONSTRUCTOR Shape(color : String, x : Integer, y : Integer)
        SET color = color
        SET x = x
        SET y = y

    METHODS
        calculateArea() : Double
        calculatePerimeter() : Double
END CLASS


CLASS Circle INHERITS Shape
    ATTRIBUTES
        radius : Double

    CONSTRUCTOR Circle(color : String, x : Integer, y : Integer, radius : Double)
        CALL Shape(color, x, y)
        SET radius = radius

    METHODS
        calculateArea() : Double
            RETURN 3.14 * radius * radius

        calculatePerimeter() : Double
            RETURN 2 * 3.14 * radius
END CLASS


CLASS Rectangle INHERITS Shape
    ATTRIBUTES
        width : Double
        height : Double

    CONSTRUCTOR Rectangle(color : String, x : Integer, y : Integer,
                          width : Double, height : Double)
        CALL Shape(color, x, y)
        SET width = width
        SET height = height

    METHODS
        calculateArea() : Double
            RETURN width * height

        calculatePerimeter() : Double
            RETURN 2 * (width + height)
END CLASS


CLASS Triangle INHERITS Shape
    ATTRIBUTES
        side1 : Double
        side2 : Double
        side3 : Double

    CONSTRUCTOR Triangle(color : String, x : Integer, y : Integer,
                         side1 : Double, side2 : Double, side3 : Double)
        CALL Shape(color, x, y)
        SET side1 = side1
        SET side2 = side2
        SET side3 = side3

    METHODS
        calculateArea() : Double
            RETURN "Use Heron's Formula"

        calculatePerimeter() : Double
            RETURN side1 + side2 + side3
END CLASS


CLASS Canvas
    ATTRIBUTES
        shapes : List<Shape>

    CONSTRUCTOR Canvas()
        CREATE empty shapes list

    METHODS
        addShape(shape : Shape)
            ADD shape TO shapes

        displayShapes()
            FOR EACH shape IN shapes
                PRINT shape
            END FOR
END CLASS


START PROGRAM

DECLARE canvas : Canvas
SET canvas = NEW Canvas()

DECLARE circle : Circle
SET circle = NEW Circle("Red", 10, 20, 5)

DECLARE rectangle : Rectangle
SET rectangle = NEW Rectangle("Blue", 30, 40, 8, 4)

DECLARE triangle : Triangle
SET triangle = NEW Triangle("Green", 50, 60, 3, 4, 5)

CALL canvas.addShape(circle)
CALL canvas.addShape(rectangle)
CALL canvas.addShape(triangle)

PRINT "Circle Area = " + circle.calculateArea()
PRINT "Rectangle Area = " + rectangle.calculateArea()
PRINT "Triangle Perimeter = " + triangle.calculatePerimeter()

CALL canvas.displayShapes()

END PROGRAM



Task 294: Employee Payroll 




PROGRAM EmployeePayroll

CLASS Employee
    ATTRIBUTES
        name : String
        id : Integer
        hireDate : String

    CONSTRUCTOR Employee(name : String, id : Integer, hireDate : String)
        SET name = name
        SET id = id
        SET hireDate = hireDate

    METHODS
        calculateMonthlyPay() : Double
END CLASS


CLASS SalariedEmployee INHERITS Employee
    ATTRIBUTES
        monthlySalary : Double

    CONSTRUCTOR SalariedEmployee(name : String, id : Integer,
                                 hireDate : String, monthlySalary : Double)
        CALL Employee(name, id, hireDate)
        SET monthlySalary = monthlySalary

    METHODS
        calculateMonthlyPay() : Double
            RETURN monthlySalary
END CLASS


CLASS HourlyEmployee INHERITS Employee
    ATTRIBUTES
        hourlyRate : Double
        hoursWorked : Double

    CONSTRUCTOR HourlyEmployee(name : String, id : Integer,
                               hireDate : String,
                               hourlyRate : Double,
                               hoursWorked : Double)
        CALL Employee(name, id, hireDate)
        SET hourlyRate = hourlyRate
        SET hoursWorked = hoursWorked

    METHODS
        calculateMonthlyPay() : Double
            RETURN hourlyRate * hoursWorked
END CLASS


CLASS CommissionEmployee INHERITS Employee
    ATTRIBUTES
        basePay : Double
        sales : Double
        commissionRate : Double

    CONSTRUCTOR CommissionEmployee(name : String, id : Integer,
                                   hireDate : String,
                                   basePay : Double,
                                   sales : Double,
                                   commissionRate : Double)
        CALL Employee(name, id, hireDate)
        SET basePay = basePay
        SET sales = sales
        SET commissionRate = commissionRate

    METHODS
        calculateMonthlyPay() : Double
            RETURN basePay + (sales * commissionRate)
END CLASS


CLASS PayrollSystem
    ATTRIBUTES
        employees : List<Employee>

    CONSTRUCTOR PayrollSystem()
        CREATE empty employees list

    METHODS
        addEmployee(employee : Employee)
            ADD employee TO employees

        displayPayroll()
            FOR EACH employee IN employees
                PRINT employee.name
                PRINT employee.calculateMonthlyPay()
            END FOR
END CLASS


START PROGRAM

DECLARE payroll : PayrollSystem
SET payroll = NEW PayrollSystem()

DECLARE emp1 : SalariedEmployee
SET emp1 = NEW SalariedEmployee("Ahmed", 101, "2023-01-10", 5000)

DECLARE emp2 : HourlyEmployee
SET emp2 = NEW HourlyEmployee("Sara", 102, "2024-03-15", 20, 160)

DECLARE emp3 : CommissionEmployee
SET emp3 = NEW CommissionEmployee("Ali", 103, "2022-08-20", 3000, 10000, 0.10)

CALL payroll.addEmployee(emp1)
CALL payroll.addEmployee(emp2)
CALL payroll.addEmployee(emp3)

CALL payroll.displayPayroll()

END PROGRAM

Task 296: Online Payment Method


CLASS PaymentMethod

    ATTRIBUTES
        amount : Double
        status : String

    CONSTRUCTOR PaymentMethod(amount : Double)
        SET this.amount = amount
        SET this.status = "pending"

    METHODS
        processPayment()
            PRINT "Processing payment"

END CLASS



CLASS CreditCardPayment INHERITS PaymentMethod

    ATTRIBUTES
        cardNumber : String
        expiryDate : String

    CONSTRUCTOR CreditCardPayment(amount : Double,cardNumber : String, expiryDate : String)

        CALL SUPER(amount)
        SET this.cardNumber = cardNumber
        SET this.expiryDate = expiryDate


    OVERRIDE METHOD processPayment()

        IF cardNumber IS VALID AND expiryDate IS VALID THEN
            SET status = "completed"
            PRINT "Credit card payment completed"
        ELSE
            PRINT "Invalid card details"
        END IF

END CLASS



CLASS BankTransferPayment INHERITS PaymentMethod

    ATTRIBUTES
        accountNumber : String

    CONSTRUCTOR BankTransferPayment(amount : Double,accountNumber : String)

        CALL SUPER(amount)
        SET this.accountNumber = accountNumber


    OVERRIDE METHOD processPayment()

        IF accountNumber IS VALID THEN
            PRINT "Processing bank transfer delay"
            SET status = "completed"
            PRINT "Bank transfer completed"
        ELSE
            PRINT "Invalid account number"
        END IF

END CLASS



CLASS WalletPayment INHERITS PaymentMethod

    ATTRIBUTES
        walletBalance : Double

    CONSTRUCTOR WalletPayment(amount : Double,walletBalance : Double)

        CALL SUPER(amount)
        SET this.walletBalance = walletBalance


    OVERRIDE METHOD processPayment()

        IF walletBalance >= amount THEN
            SET status = "completed"
            PRINT "Wallet payment completed"
        ELSE
            PRINT "Insufficient wallet balance"
        END IF

END CLASS



CLASS Item

    ATTRIBUTES
        itemName : String
        price : Double

    CONSTRUCTOR Item(itemName : String, price : Double)

        SET this.itemName = itemName
        SET this.price = price

END CLASS



CLASS Order

    ATTRIBUTES
        items : List(Item)
        payment : PaymentMethod

    CONSTRUCTOR Order()

        CREATE empty items list
        SET payment = NULL


    METHODS

        addItem(item : Item)

            ADD item TO items


        checkout(paymentMethod : PaymentMethod)

            SET payment = paymentMethod
            CALL payment.processPayment()


    METHOD calculateTotal() 

    DECLARE total : Double
    DECLARE i : Integer

    SET total = 0
    SET i = 0

    WHILE i < size of items

        SET total = total + items[i].price

        SET i = i + 1

    END WHILE

    RETURN total

END METHOD
END CLASS
