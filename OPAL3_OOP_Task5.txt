CLASS Shape
    ATTRIBUTES
        color : String
        x : Integer
        y : Integer

    CONSTRUCTOR Shape(color : String, x : Integer, y : Integer)
        SET color = color
        SET x = x
        SET y = y

    METHODS
        calculateArea() : Double
        calculatePerimeter() : Double
END CLASS


CLASS Circle INHERITS Shape
    ATTRIBUTES
        radius : Double

    CONSTRUCTOR Circle(color : String, x : Integer, y : Integer, radius : Double)
        CALL Shape(color, x, y)
        SET radius = radius

    METHODS
        calculateArea() : Double
            RETURN 3.14 * radius * radius

        calculatePerimeter() : Double
            RETURN 2 * 3.14 * radius
END CLASS


CLASS Rectangle INHERITS Shape
    ATTRIBUTES
        width : Double
        height : Double

    CONSTRUCTOR Rectangle(color : String, x : Integer, y : Integer,
                          width : Double, height : Double)
        CALL Shape(color, x, y)
        SET width = width
        SET height = height

    METHODS
        calculateArea() : Double
            RETURN width * height

        calculatePerimeter() : Double
            RETURN 2 * (width + height)
END CLASS


CLASS Triangle INHERITS Shape
    ATTRIBUTES
        side1 : Double
        side2 : Double
        side3 : Double

    CONSTRUCTOR Triangle(color : String, x : Integer, y : Integer,
                         side1 : Double, side2 : Double, side3 : Double)
        CALL Shape(color, x, y)
        SET side1 = side1
        SET side2 = side2
        SET side3 = side3

    METHODS
        calculateArea() : Double
            RETURN "Use Heron's Formula"

        calculatePerimeter() : Double
            RETURN side1 + side2 + side3
END CLASS


CLASS Canvas
    ATTRIBUTES
        shapes : List<Shape>

    CONSTRUCTOR Canvas()
        CREATE empty shapes list

    METHODS
        addShape(shape : Shape)
            ADD shape TO shapes

        displayShapes()
            FOR EACH shape IN shapes
                PRINT shape
            END FOR
END CLASS


START PROGRAM

DECLARE canvas : Canvas
SET canvas = NEW Canvas()

DECLARE circle : Circle
SET circle = NEW Circle("Red", 10, 20, 5)

DECLARE rectangle : Rectangle
SET rectangle = NEW Rectangle("Blue", 30, 40, 8, 4)

DECLARE triangle : Triangle
SET triangle = NEW Triangle("Green", 50, 60, 3, 4, 5)

CALL canvas.addShape(circle)
CALL canvas.addShape(rectangle)
CALL canvas.addShape(triangle)

PRINT "Circle Area = " + circle.calculateArea()
PRINT "Rectangle Area = " + rectangle.calculateArea()
PRINT "Triangle Perimeter = " + triangle.calculatePerimeter()

CALL canvas.displayShapes()

END PROGRAM

CLASS CreditCardPayment INHERITS PaymentMethod

    ATTRIBUTES
        cardNumber : String
        expiryDate : String

        CALL SUPER(amount)
        SET this.cardNumber = cardNumber

    OVERRIDE METHOD processPayment()

        IF cardNumber IS VALID AND expiryDate IS VALID THEN
            SET status = "completed"

        END IF

CLASS Order

    ATTRIBUTES
        items : List(Item)
        payment : PaymentMethod

    CONSTRUCTOR Order()

        CREATE empty items list
        SET payment = NULL


    METHODS

        addItem(item : Item)

            ADD item TO items


        checkout(paymentMethod : PaymentMethod)

            SET payment = paymentMethod
            CALL payment.processPayment()


    METHOD calculateTotal() 

    DECLARE total : Double
    DECLARE i : Integer

    SET total = 0
    SET i = 0

    WHILE i < size of items

        SET total = total + items[i].price

        SET i = i + 1

    END WHILE

    RETURN total

END METHOD
END CLASS

CLASS Card

    ATTRIBUTES
        cardNumber : String
        expiryDate : String
        status : String
        chip : Chip

    CONSTRUCTOR Card(cardNumber : String,expiryDate : String,chip : Chip)

        SET this.cardNumber = cardNumber
        SET this.expiryDate = expiryDate
        SET this.status = "active"
        SET this.chip = chip


    METHODS

        authorisePurchase(amount : Double) : Boolean

            RETURN FALSE

END CLASS



CLASS DebitCard INHERITS Card

    ATTRIBUTES
        accountBalance : Double


    CONSTRUCTOR DebitCard(cardNumber : String,expiryDate : String,chip : Chip,accountBalance : Double)

        CALL SUPER(cardNumber, expiryDate, chip)

        SET this.accountBalance = accountBalance


    OVERRIDE METHOD authorisePurchase(amount : Double) : Boolean

        IF accountBalance >= amount THEN

            SET accountBalance = accountBalance - amount

            PRINT "Debit purchase approved"

            RETURN TRUE

        ELSE

            PRINT "Insufficient account balance"

            RETURN FALSE

        END IF

END CLASS



CLASS CreditCard INHERITS Card

    ATTRIBUTES
        creditLimit : Double
        unpaidPurchases : List(Double)


    CONSTRUCTOR CreditCard(cardNumber : String,expiryDate : String,chip : Chip,creditLimit : Double)

        CALL SUPER(cardNumber, expiryDate, chip)

        SET this.creditLimit = creditLimit

        CREATE empty unpaidPurchases list



    METHODS


        OVERRIDE METHOD authorisePurchase(amount : Double) : Boolean

            IF calculateOutstandingBalance() + amount <= creditLimit THEN

                ADD amount TO unpaidPurchases

                PRINT "Credit purchase approved"

                RETURN TRUE

            ELSE

                PRINT "Credit limit exceeded"

                RETURN FALSE

            END IF



        calculateOutstandingBalance() : Double

            DECLARE total : Double

            SET total = 0

            SET i = 0

            WHILE i < size of unpaidPurchases

                SET total = total + unpaidPurchases[i]

                SET i = i + 1

            END WHILE


            RETURN total

END CLASS



CLASS Chip

    ATTRIBUTES
        encryptedSecurityData : String

    CONSTRUCTOR Chip(encryptedSecurityData : String)

        SET this.encryptedSecurityData = encryptedSecurityData

END CLASS
