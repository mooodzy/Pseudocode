CLASS Shape (ABSTRACT)

    ATTRIBUTES
        color : String
        x : Double
        y : Double


    CONSTRUCTOR Shape(color : String, x : Double, y : Double)

        SET this.color = color
        SET this.x = x
        SET this.y = y


    ABSTRACT METHOD calculateArea() : Double

    ABSTRACT METHOD calculatePerimeter() : Double


END CLASS



CLASS Circle INHERITS Shape

    ATTRIBUTES
        radius : Double


    CONSTRUCTOR Circle(color : String, x : Double, y : Double, radius : Double)

        CALL SUPER(color, x, y)

        SET this.radius = radius


    OVERRIDE METHOD calculateArea() : Double

        RETURN 3.14 * radius * radius


    OVERRIDE METHOD calculatePerimeter() : Double

        RETURN 2 * 3.14 * radius


END CLASS



CLASS Rectangle INHERITS Shape

    ATTRIBUTES
        width : Double
        height : Double


    CONSTRUCTOR Rectangle(color : String, x : Double, y : Double, width : Double, height : Double)

        CALL SUPER(color, x, y)

        SET this.width = width
        SET this.height = height


    OVERRIDE METHOD calculateArea() : Double

        RETURN width * height


    OVERRIDE METHOD calculatePerimeter() : Double

        RETURN 2 * (width + height)


END CLASS



CLASS Triangle INHERITS Shape

    ATTRIBUTES
        side1 : Double
        side2 : Double
        side3 : Double


    CONSTRUCTOR Triangle(color : String, x : Double, y : Double, side1 : Double, side2 : Double, side3 : Double)

        CALL SUPER(color, x, y)

        SET this.side1 = side1
        SET this.side2 = side2
        SET this.side3 = side3


    OVERRIDE METHOD calculateArea() : Double

        SET s = (side1 + side2 + side3) / 2

        RETURN sqrt(s * (s - side1) * (s - side2) * (s - side3))


    OVERRIDE METHOD calculatePerimeter() : Double

        RETURN side1 + side2 + side3


END CLASS



CLASS Canvas

    ATTRIBUTES
        shapes : List<Shape>


    CONSTRUCTOR Canvas()

        CREATE empty shapes list


    METHOD addShape(shape : Shape)

        ADD shape TO shapes



    METHOD displayShapes()

        SET i = 0

        WHILE i < size of shapes

            PRINT "Color: " + shapes[i].color

            PRINT "Area: " + shapes[i].calculateArea()

            PRINT "Perimeter: " + shapes[i].calculatePerimeter()

            SET i = i + 1

        END WHILE


END CLASS



START PROGRAM


DECLARE canvas : Canvas

SET canvas = NEW Canvas()



DECLARE circle : Circle

SET circle = NEW Circle("Red", 10, 20, 5)



DECLARE rectangle : Rectangle

SET rectangle = NEW Rectangle("Blue", 30, 40, 8, 4)



DECLARE triangle : Triangle

SET triangle = NEW Triangle("Green", 50, 60, 3, 4, 5)



CALL canvas.addShape(circle)

CALL canvas.addShape(rectangle)

CALL canvas.addShape(triangle)



PRINT "Circle Area = "

PRINT circle.calculateArea()



PRINT "Rectangle Area = "

PRINT rectangle.calculateArea()



PRINT "Triangle Perimeter = "

PRINT triangle.calculatePerimeter()

                                 hireDate : String, monthlySalary : Double)

        CALL Employee(name, id, hireDate)
        SET hourlyRate = hourlyRate
        SET hoursWorked = hoursWorked

    METHODS

    ATTRIBUTES
        basePay : Double
        sales : Double
        commissionRate : Double

        CALL Employee(name, id, hireDate)
        SET basePay = basePay
        SET sales = sales
        SET commissionRate = commissionRate

    METHODS
        calculateMonthlyPay() : Double

CLASS Card

    ATTRIBUTES
        cardNumber : String
        expiryDate : String
        status : String
        chip : Chip

    CONSTRUCTOR Card(cardNumber : String,expiryDate : String,chip : Chip)

        SET this.cardNumber = cardNumber
        SET this.expiryDate = expiryDate
        SET this.status = "active"
        SET this.chip = chip


    METHODS

        authorisePurchase(amount : Double) : Boolean

            RETURN FALSE

END CLASS



CLASS DebitCard INHERITS Card

    ATTRIBUTES
        accountBalance : Double


    CONSTRUCTOR DebitCard(cardNumber : String,expiryDate : String,chip : Chip,accountBalance : Double)

        CALL SUPER(cardNumber, expiryDate, chip)

        SET this.accountBalance = accountBalance


    OVERRIDE METHOD authorisePurchase(amount : Double) : Boolean

        IF accountBalance >= amount THEN

            SET accountBalance = accountBalance - amount

            PRINT "Debit purchase approved"

            RETURN TRUE

        ELSE

            PRINT "Insufficient account balance"

            RETURN FALSE

        END IF

END CLASS



CLASS CreditCard INHERITS Card

    ATTRIBUTES
        creditLimit : Double
        unpaidPurchases : List(Double)


    CONSTRUCTOR CreditCard(cardNumber : String,expiryDate : String,chip : Chip,creditLimit : Double)

        CALL SUPER(cardNumber, expiryDate, chip)

        SET this.creditLimit = creditLimit

        CREATE empty unpaidPurchases list



    METHODS


        OVERRIDE METHOD authorisePurchase(amount : Double) : Boolean

            IF calculateOutstandingBalance() + amount <= creditLimit THEN

                ADD amount TO unpaidPurchases

                PRINT "Credit purchase approved"

                RETURN TRUE

            ELSE

                PRINT "Credit limit exceeded"

                RETURN FALSE

            END IF



        calculateOutstandingBalance() : Double

            DECLARE total : Double

            SET total = 0

            SET i = 0

            WHILE i < size of unpaidPurchases

                SET total = total + unpaidPurchases[i]

                SET i = i + 1

            END WHILE


            RETURN total

END CLASS

    ATTRIBUTES
        encryptedSecurityData : String

    CONSTRUCTOR Chip(encryptedSecurityData : String)

        SET this.encryptedSecurityData = encryptedSecurityData

END CLASS

        storage : StorageUnit


    CONSTRUCTOR SmartCamera(name : String,roomLocation : String,status : Boolean,recording : Boolean,
    storage : StorageUnit)

        CALL SUPER(name, roomLocation, status)

        SET this.recording = recording
        SET this.storage = storage


    OVERRIDE METHOD getPowerUsage() : Double

        IF recording = TRUE THEN

            RETURN 10.0

        ELSE

            RETURN 1.0

        END IF


    METHODS

        startRecording()

            IF storage.isFull() = FALSE THEN

                SET recording = TRUE

                PRINT "Camera is recording"

            ELSE

                PRINT "Storage is full"

            END IF


END CLASS



CLASS StorageUnit

    ATTRIBUTES
        usedSpace : Double
        totalSpace : Double


    CONSTRUCTOR StorageUnit(usedSpace : Double,totalSpace : Double)

        SET this.usedSpace = usedSpace
        SET this.totalSpace = totalSpace


    METHODS

        isFull() : Boolean

            IF usedSpace >= totalSpace THEN

                RETURN TRUE

            ELSE

                RETURN FALSE

            END IF


END CLASS


CLASS Room

    ATTRIBUTES
        roomName : String
        devices : List(Device)


    CONSTRUCTOR Room(roomName : String)

        SET this.roomName = roomName

        CREATE empty devices list


    METHODS

        addDevice(device : Device)

            ADD device TO devices


        displayDevices()

            SET i = 0

            WHILE i < size of devices

                PRINT devices[i].name

                PRINT devices[i].getPowerUsage()

                SET i = i + 1

            END WHILE

END CLASS
