PROGRAM Bookshop


CLASS Genre

    ATTRIBUTES
        PRIVATE genreName : String

    CONSTRUCTOR Genre(name : String)
        SET this.genreName TO name
    END CONSTRUCTOR


    METHOD getGenreName() : String
        RETURN genreName
    END METHOD

END CLASS



CLASS Book

    ATTRIBUTES
        PRIVATE title : String
        PRIVATE author : String
        PRIVATE price : Double
        PRIVATE genre : Genre


    CONSTRUCTOR Book(title : String, author : String, price : Double, genre : Genre)

        SET this.title TO title
        SET this.author TO author
        SET this.price TO price
        SET this.genre TO genre

    END CONSTRUCTOR


    METHOD getTitle() : String
        RETURN title
    END METHOD


    METHOD getAuthor() : String
        RETURN author
    END METHOD


    METHOD getPrice() : Double
        RETURN price
    END METHOD


    METHOD getGenre() : Genre
        RETURN genre
    END METHOD

END CLASS




CLASS Sale

    ATTRIBUTES
        PRIVATE book : Book
        PRIVATE soldPrice : Double


    CONSTRUCTOR Sale(book : Book, price : Double)

        SET this.book TO book
        SET this.soldPrice TO price

    END CONSTRUCTOR


    METHOD getSoldPrice() : Double
        RETURN soldPrice
    END METHOD


    METHOD getBook() : Book
        RETURN book
    END METHOD

END CLASS




CLASS Customer

    ATTRIBUTES
        PRIVATE customerID : Integer
        PRIVATE customerName : String
        PRIVATE purchaseHistory : List<Sale>


    CONSTRUCTOR Customer(id : Integer, name : String)

        SET this.customerID TO id
        SET this.customerName TO name
        SET purchaseHistory TO Empty List

    END CONSTRUCTOR



    METHOD buyBook(book : Book)

        CREATE sale1 OF Sale(book, book.getPrice())

        ADD sale1 TO purchaseHistory

    END METHOD



    METHOD getTotalPurchases() : Double

        SET total : Double TO 0


        FOR EACH sale IN purchaseHistory

            SET total TO total + sale.getSoldPrice()

        END FOR


        RETURN total

    END METHOD



    METHOD getCustomerName() : String
        RETURN customerName
    END METHOD


    METHOD getCustomerID() : Integer
        RETURN customerID
    END METHOD

END CLASS




START PROGRAM


    CREATE genre1 OF Genre("Fiction")


    CREATE book1 OF Book("Harry Potter", "J.K. Rowling",50.0, genre1 )


    CREATE customer1 OF Customer(101,"Ali")


    CALL customer1.buyBook(book1)


    DISPLAY customer1.getTotalPurchases()


END PROGRAM


Task 289: Online Exam Platform pseudocode


CLASS Student

    ATTRIBUTES
        PRIVATE studentID : Integer
        PRIVATE studentName : String
        PRIVATE attempts : List<Attempt>


    CONSTRUCTOR Student(id : Integer, name : String)

        SET this.studentID TO id
        SET this.studentName TO name
        SET attempts TO Empty List

    END CONSTRUCTOR



    METHOD startExam(exam : Exam) : Attempt

        CREATE attempt1 OF Attempt(exam, this)

        ADD attempt1 TO attempts

        RETURN attempt1

    END METHOD



    METHOD getStudentName() : String
        RETURN studentName
    END METHOD



    METHOD getStudentID() : Integer
        RETURN studentID
    END METHOD

END CLASS




CLASS Question

    ATTRIBUTES
        PRIVATE questionText : String
        PRIVATE correctAnswer : String
        PRIVATE marks : Integer


    CONSTRUCTOR Question(text : String, answer : String, marks : Integer)

        SET questionText TO text
        SET correctAnswer TO answer
        SET this.marks TO marks

    END CONSTRUCTOR



    METHOD getCorrectAnswer() : String
        RETURN correctAnswer
    END METHOD



    METHOD getMarks() : Integer
        RETURN marks
    END METHOD



    METHOD getQuestionText() : String
        RETURN questionText
    END METHOD

END CLASS




CLASS Exam

    ATTRIBUTES
        PRIVATE examID : Integer
        PRIVATE questions : List<Question>
        PRIVATE timeLimit : Integer


    CONSTRUCTOR Exam(id : Integer, limit : Integer)

        SET examID TO id
        SET timeLimit TO limit
        SET questions TO Empty List

    END CONSTRUCTOR



    METHOD addQuestion(q : Question)

        ADD q TO questions

    END METHOD



    METHOD getQuestions() : List<Question>

        RETURN questions

    END METHOD



    METHOD getTimeLimit() : Integer

        RETURN timeLimit

    END METHOD

END CLASS




CLASS Attempt

    ATTRIBUTES
        PRIVATE exam : Exam
        PRIVATE student : Student
        PRIVATE answers : List<String>
        PRIVATE score : Integer
        PRIVATE startTime : Integer


    CONSTRUCTOR Attempt(exam : Exam, student : Student)

        SET this.exam TO exam
        SET this.student TO student
        SET answers TO Empty List
        SET score TO 0
        SET startTime TO SYSTEM.CURRENT_TIME()

    END CONSTRUCTOR



    METHOD addAnswer(answer : String)

        IF SYSTEM.CURRENT_TIME() - startTime > exam.getTimeLimit() THEN

            DISPLAY "Time is over - no more answers allowed"

        ELSE IF answers.size >= exam.getQuestions().size THEN

            DISPLAY "All questions are already answered"

        ELSE

            ADD answer TO answers

        END IF

    END METHOD



    METHOD calculateScore() : Integer

        SET score TO 0

        FOR i FROM 0 TO answers.size - 1

            IF answers[i] = exam.getQuestions()[i].getCorrectAnswer() THEN

                SET score TO score + exam.getQuestions()[i].getMarks()

            END IF

        END FOR


        RETURN score

    END METHOD



    METHOD getScore() : Integer

        RETURN score

    END METHOD

END CLASS




START PROGRAM


    CREATE student1 OF Student(101, "Ali")


    CREATE exam1 OF Exam(201, 60)



    CREATE q1 OF Question("2 + 2 = ?", "4", 5)

    CREATE q2 OF Question("5 + 3 = ?", "8", 5)



    CALL exam1.addQuestion(q1)

    CALL exam1.addQuestion(q2)



    CREATE attempt1

    SET attempt1 TO student1.startExam(exam1)



    CALL attempt1.addAnswer("4")

    CALL attempt1.addAnswer("8")



    DISPLAY attempt1.calculateScore()


END PROGRAM

Task 291: Co-working Space 

BEGIN

CLASS MembershipPlan
    ATTRIBUTES
        PRIVATE planName : String
        PRIVATE monthlyQuota : Integer

    CONSTRUCTOR MembershipPlan(name, quota)
        SET planName TO name
        SET monthlyQuota TO quota
    END CONSTRUCTOR
END CLASS


CLASS Member
    ATTRIBUTES
        PRIVATE memberID : Integer
        PRIVATE memberName : String
        PRIVATE plan : MembershipPlan
        PRIVATE bookings : List of Booking

    CONSTRUCTOR Member(id, name, plan)
        SET memberID TO id
        SET memberName TO name
        SET this.plan TO plan
        SET bookings TO Empty List
    END CONSTRUCTOR

    METHOD addBooking(booking)
        ADD booking TO bookings
    END METHOD

    METHOD getRemainingHours()
        SET usedHours TO 0

        REPEAT FROM i = 0 TO bookings.size - 1
            SET usedHours TO usedHours + bookings[i].duration
        END REPEAT

        RETURN plan.monthlyQuota - usedHours
    END METHOD
END CLASS


CLASS Room
    ATTRIBUTES
        PRIVATE roomID : Integer
        PRIVATE roomName : String
        PRIVATE bookings : List of Booking

    CONSTRUCTOR Room(id, name)
        SET roomID TO id
        SET roomName TO name
        SET bookings TO Empty List
    END CONSTRUCTOR

    METHOD isAvailable(startTime, endTime)
        REPEAT FROM i = 0 TO bookings.size - 1
            IF startTime < bookings[i].endTime AND endTime > bookings[i].startTime THEN
                RETURN False
            END IF
        END REPEAT

        RETURN True
    END METHOD

    METHOD addBooking(booking)
        ADD booking TO bookings
    END METHOD
END CLASS


CLASS Booking
    ATTRIBUTES
        PRIVATE bookingID : Integer
        PRIVATE member : Member
        PRIVATE room : Room
        PRIVATE startTime : Integer
        PRIVATE endTime : Integer
        PRIVATE duration : Integer

    CONSTRUCTOR Booking(id, member, room, startTime, endTime)
        SET bookingID TO id
        SET this.member TO member
        SET this.room TO room
        SET this.startTime TO startTime
        SET this.endTime TO endTime
        SET duration TO endTime - startTime
    END CONSTRUCTOR
END CLASS


START
BEGIN

    CREATE OBJECT plan1 OF MembershipPlan( "Standard", 20)

    CREATE OBJECT member1 OF Member(101, "Ali", plan1)

    CREATE OBJECT room1 OF Room(201, "Meeting Room A")

    IF member1.getRemainingHours() >= 2 AND room1.isAvailable(9, 11) THEN

        CREATE OBJECT booking1 OF Booking(301, member1, room1, 9, 11)

        member1.addBooking(booking1)
        room1.addBooking(booking1)

        DISPLAY "Booking Successful"

    ELSE

        DISPLAY "Booking Rejected"

    END IF

    DISPLAY member1.getRemainingHours()

END
STOP

END PROGRAM

BEGIN

CLASS Client

        PRIVATE clientID : Integer

        SET clientName TO name
        SET this.phone TO phone
        SET this.email TO email
        SET sessions TO Empty List
    END CONSTRUCTOR

    METHOD addSession(session)
        ADD session TO sessions
    END METHOD

CLASS Session
    ATTRIBUTES
        PRIVATE sessionID : Integer
        PRIVATE client : Client
        PRIVATE sessionDate : String
        PRIVATE duration : Integer
        PRIVATE workoutType : String

    CONSTRUCTOR Session(id, client, date, duration, workout)
        SET sessionID TO id
        SET this.client TO client
        SET sessionDate TO date
        SET this.duration TO duration
        SET workoutType TO workout
    END CONSTRUCTOR
END CLASS


CLASS Trainer
    ATTRIBUTES
        PRIVATE trainerID : Integer
        PRIVATE trainerName : String
        PRIVATE sessions : List of Session

    CONSTRUCTOR Trainer(id, name)
        SET trainerID TO id
        SET trainerName TO name
        SET sessions TO Empty List
    END CONSTRUCTOR

    METHOD addSession(session)
        ADD session TO sessions
    END METHOD

    METHOD displaySessions()
        REPEAT FROM i = 0 TO sessions.size - 1
            DISPLAY sessions[i]
        END REPEAT
    END METHOD
END CLASS

    CREATE OBJECT client1 OF Client(101, "Ali", "98765432", "ali@email.com")

    CREATE OBJECT trainer1 OF Trainer(201, "Ahmed")

    trainer1.addSession(session1)

    DISPLAY "Session Booked Successfully"
