CLASS Genre
    ATTRIBUTES
        PRIVATE genreName : String

    CONSTRUCTOR Genre(name)
        SET genreName TO name
    END CONSTRUCTOR
END CLASS

CLASS Book
    ATTRIBUTES
        PRIVATE title : String
        PRIVATE author : String
        PRIVATE price : Double
        PRIVATE genre : Genre

    CONSTRUCTOR Book(title, author, price, genre)
        SET this.title TO title
        SET this.author TO author
        SET this.price TO price
        SET this.genre TO genre
    END CONSTRUCTOR
END CLASS


CLASS Sale
    ATTRIBUTES
        PRIVATE book : Book
        PRIVATE soldPrice : Double

    CONSTRUCTOR Sale(book, price)
        SET this.book TO book
        SET this.soldPrice TO price
    END CONSTRUCTOR
END CLASS



CLASS Customer
    ATTRIBUTES
        PRIVATE customerID : Integer
        PRIVATE customerName : String
        PRIVATE purchaseHistory : List of Sale

    CONSTRUCTOR Customer(id, name)
        SET customerID TO id
        SET customerName TO name
        SET purchaseHistory TO Empty List
    END CONSTRUCTOR

    METHOD buyBook(book)
        CREATE sale1 OF Sale(book, book.price)
        ADD sale1 TO purchaseHistory
    END METHOD

    METHOD getTotalPurchases()
        SET total TO 0

        REPEAT FROM i = 0 TO purchaseHistory.size - 1
            SET total TO total + purchaseHistory[i].soldPrice
        END REPEAT

        RETURN total
    END METHOD
END CLASS



START
BEGIN

    CREATE OBJECT genre1 OF Genre("Fiction")

    CREATE OBJECT book1 OF Book("Harry Potter", "J.K. Rowling", 50, genre1)

    CREATE OBJECT customer1 OF Customer(101, "Ali")

    customer1.buyBook(book1)

    DISPLAY customer1.getTotalPurchases()

END
STOP

END PROGRAM


Task 289: Online Exam Platform 
CLASS Student
    ATTRIBUTES
        PRIVATE studentID : Integer
        PRIVATE studentName : String
        PRIVATE attempts : List of Attempt

    CONSTRUCTOR Student(id, name)
        SET studentID TO id
        SET studentName TO name
        SET attempts TO Empty List
    END CONSTRUCTOR

    METHOD startExam(exam)
        CREATE attempt1 OF Attempt(exam, this)
        ADD attempt1 TO attempts
        RETURN attempt1
    END METHOD
END CLASS


CLASS Question
    ATTRIBUTES
        PRIVATE questionText : String
        PRIVATE correctAnswer : String
        PRIVATE marks : Integer

    CONSTRUCTOR Question(text, answer, marks)
        SET questionText TO text
        SET correctAnswer TO answer
        SET this.marks TO marks
    END CONSTRUCTOR
END CLASS


CLASS Exam
    ATTRIBUTES
        PRIVATE examID : Integer
        PRIVATE questions : List of Question
        PRIVATE timeLimit : Integer   // in minutes

    CONSTRUCTOR Exam(id, timeLimit)
        SET examID TO id
        SET this.timeLimit TO timeLimit
        SET questions TO Empty List
    END CONSTRUCTOR

    METHOD addQuestion(q)
        ADD q TO questions
    END METHOD
END CLASS


CLASS Attempt
    ATTRIBUTES
        PRIVATE exam : Exam
        PRIVATE student : Student
        PRIVATE answers : List of String
        PRIVATE score : Integer
        PRIVATE startTime : Integer

    CONSTRUCTOR Attempt(exam, student)
        SET this.exam TO exam
        SET this.student TO student
        SET answers TO Empty List
        SET score TO 0
        SET startTime TO CurrentTime
    END CONSTRUCTOR

    METHOD addAnswer(answer)
        IF CurrentTime - startTime > exam.timeLimit THEN
            DISPLAY "Time is over - no more answers allowed"
        ELSE
            ADD answer TO answers
        END IF
    END METHOD

    METHOD calculateScore()
        SET score TO 0

        REPEAT FROM i = 0 TO exam.questions.size - 1
            IF answers[i] = exam.questions[i].correctAnswer THEN
                SET score TO score + exam.questions[i].marks
            END IF
        END REPEAT

        RETURN score
    END METHOD
END CLASS


START
BEGIN

    CREATE OBJECT student1 OF Student(101, "Ali")

    CREATE OBJECT exam1 OF Exam(201, 60)

    CREATE OBJECT q1 OF Question("2 + 2 = ?", "4", 5)
    CREATE OBJECT q2 OF Question("5 + 3 = ?", "8", 5)

    exam1.addQuestion(q1)
    exam1.addQuestion(q2)

    attempt1 = student1.startExam(exam1)

    attempt1.addAnswer("4")
    attempt1.addAnswer("8")

    DISPLAY attempt1.calculateScore()

END
STOP
END PROGRAM
