CLASS Instrument (ABSTRACT)

    ATTRIBUTES

    CONSTRUCTOR Instrument(brand : String, price : Double)

        SET this.brand = brand

        SET this.price = price



    METHOD getBrand() : String

        RETURN brand



    METHOD getPrice() : Double

        RETURN price



    ABSTRACT METHOD describePlayStyle() : String


END CLASS

    ATTRIBUTES
        numberOfStrings : Integer

        SET this.numberOfStrings = numberOfStrings

    ATTRIBUTES
        numberOfKeys : Integer


    CONSTRUCTOR Keyboard(brand : String, price : Double, numberOfKeys : Integer)

        CALL SUPER(brand, price)

        SET this.numberOfKeys = numberOfKeys



    OVERRIDE METHOD describePlayStyle() : String

        RETURN "Pressed"


END CLASS



CLASS Store

    ATTRIBUTES
        instruments : List<Instrument>


    CONSTRUCTOR Store()

        CREATE empty instruments list



    METHOD addInstrument(instrument : Instrument)

        ADD instrument TO instruments



    METHOD displayInstruments()

        SET i = 0


        WHILE i < size of instruments

            PRINT "Brand: " + instruments[i].getBrand()

            PRINT "Price: " + instruments[i].getPrice()

            PRINT "Play Style: " + instruments[i].describePlayStyle()


            SET i = i + 1


        END WHILE


END CLASS



START PROGRAM


DECLARE store1 : Store

SET store1 = NEW Store()



DECLARE guitar1 : Guitar

SET guitar1 = NEW Guitar("Yamaha", 500.0, 6)



DECLARE drum1 : DrumKit

SET drum1 = NEW DrumKit("Pearl", 800.0, 5)



DECLARE keyboard1 : Keyboard

SET keyboard1 = NEW Keyboard("Casio", 300.0, 61)



CALL store1.addInstrument(guitar1)

CALL store1.addInstrument(drum1)

CALL store1.addInstrument(keyboard1)



CALL store1.displayInstruments()


END PROGRAM



Task 304: Fitness App Workouts



CLASS Workout (ABSTRACT)

    ATTRIBUTES
        date : String
        duration : Double


    CONSTRUCTOR Workout(date : String, duration : Double)

        SET this.date = date

        SET this.duration = duration



    METHOD getDate() : String

        RETURN date



    METHOD getDuration() : Double

        RETURN duration



    ABSTRACT METHOD calculateCalories() : Double


END CLASS



CLASS RunningWorkout INHERITS Workout

    ATTRIBUTES
        distance : Double
        pace : Double


    CONSTRUCTOR RunningWorkout(date : String, duration : Double, distance : Double, pace : Double)

        CALL SUPER(date, duration)

        SET this.distance = distance

        SET this.pace = pace



    OVERRIDE METHOD calculateCalories() : Double

        RETURN distance * pace * 10


END CLASS



CLASS CyclingWorkout INHERITS Workout

    ATTRIBUTES
        distance : Double
        resistanceLevel : Integer


    CONSTRUCTOR CyclingWorkout(date : String, duration : Double, distance : Double, resistanceLevel : Integer)

        CALL SUPER(date, duration)

        SET this.distance = distance

        SET this.resistanceLevel = resistanceLevel



    OVERRIDE METHOD calculateCalories() : Double

        RETURN distance * resistanceLevel * 8


END CLASS



CLASS WeightTrainingWorkout INHERITS Workout

    ATTRIBUTES
        sets : Integer
        reps : Integer


    CONSTRUCTOR WeightTrainingWorkout(date : String, duration : Double, sets : Integer, reps : Integer)

        CALL SUPER(date, duration)

        SET this.sets = sets

        SET this.reps = reps



    OVERRIDE METHOD calculateCalories() : Double

        RETURN sets * reps * 2


END CLASS



CLASS FitnessLog

    ATTRIBUTES
        workouts : List<Workout>


    CONSTRUCTOR FitnessLog()

        CREATE empty workouts list



    METHOD addWorkout(workout : Workout)

        ADD workout TO workouts



    METHOD displayWorkouts()

        FOR EACH workout IN workouts

            PRINT "Date: " + workout.getDate()

            PRINT "Duration: " + workout.getDuration()

            PRINT "Calories: " + workout.calculateCalories()


        END FOR


END CLASS

CLASS Bill (ABSTRACT)

    ATTRIBUTES
        billingPeriod : String
        dueDate : String
        meterReading : MeterReading


    CONSTRUCTOR Bill(billingPeriod : String, dueDate : String, meterReading : MeterReading)

        SET this.billingPeriod = billingPeriod

        SET this.dueDate = dueDate

        SET this.meterReading = meterReading



    METHOD getBillingPeriod() : String

        RETURN billingPeriod



    ABSTRACT METHOD calculateAmountOwed() : Double


END CLASS



CLASS ElectricityBill INHERITS Bill

    ATTRIBUTES
        ratePerUnit : Double


    CONSTRUCTOR ElectricityBill(billingPeriod : String, dueDate : String, meterReading : MeterReading, ratePerUnit : Double)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.ratePerUnit = ratePerUnit



    OVERRIDE METHOD calculateAmountOwed() : Double

        RETURN meterReading.getConsumption() * ratePerUnit


END CLASS



CLASS WaterBill INHERITS Bill

    ATTRIBUTES
        flatRate : Double
        standingCharge : Double


    CONSTRUCTOR WaterBill(billingPeriod : String, dueDate : String, meterReading : MeterReading, flatRate : Double, standingCharge : Double)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.flatRate = flatRate

        SET this.standingCharge = standingCharge



    OVERRIDE METHOD calculateAmountOwed() : Double

        RETURN (meterReading.getConsumption() * flatRate) + standingCharge


END CLASS



CLASS GasBill INHERITS Bill

    ATTRIBUTES
        ratePerUnit : Double
        seasonalSurcharge : Double
        month : String


    CONSTRUCTOR GasBill(billingPeriod : String, dueDate : String, meterReading : MeterReading, ratePerUnit : Double, seasonalSurcharge : Double, month : String)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.ratePerUnit = ratePerUnit

        SET this.seasonalSurcharge = seasonalSurcharge

        SET this.month = month



    OVERRIDE METHOD calculateAmountOwed() : Double

        DECLARE amount : Double

        SET amount = meterReading.getConsumption() * ratePerUnit


        IF month = "December" OR month = "January" OR month = "February" THEN

            SET amount = amount + seasonalSurcharge

        END IF


        RETURN amount


END CLASS



CLASS MeterReading

    ATTRIBUTES
        startReading : Double
        endReading : Double


    CONSTRUCTOR MeterReading(startReading : Double, endReading : Double)

        SET this.startReading = startReading

        SET this.endReading = endReading



    METHOD getConsumption() : Double

        RETURN endReading - startReading


END CLASS



CLASS CustomerAccount

    ATTRIBUTES
        bills : List<Bill>


    CONSTRUCTOR CustomerAccount()

        CREATE empty bills list



    METHOD addBill(bill : Bill)

        ADD bill TO bills



    METHOD displayBills()

        FOR EACH bill IN bills

            PRINT "Billing Period: " + bill.getBillingPeriod()

            PRINT "Amount Owed: " + bill.calculateAmountOwed()


        END FOR


END CLASS



START PROGRAM

Task 307: Ride-Sharing Vehicle Tiers  

CLASS Ride (ABSTRACT)

    ATTRIBUTES
        pickupLocation : String
        dropOffLocation : String
        distance : Double
        route : Route


    CONSTRUCTOR Ride(pickupLocation : String,dropOffLocation : String,distance : Double, route : Route)

        SET this.pickupLocation = pickupLocation

        SET this.dropOffLocation = dropOffLocation

        SET this.distance = distance

        SET this.route = route



    METHOD getDistance() : Double

        RETURN distance



    ABSTRACT METHOD calculateFare() : Double


END CLASS



CLASS StandardRide INHERITS Ride

    ATTRIBUTES
        ratePerKilometer : Double


    CONSTRUCTOR StandardRide(pickupLocation : String,dropOffLocation : String,distance : Double,route : Route,ratePerKilometer : Double)

        CALL SUPER(pickupLocation, dropOffLocation, distance, route)

        SET this.ratePerKilometer = ratePerKilometer



    OVERRIDE METHOD calculateFare() : Double

        RETURN distance * ratePerKilometer


END CLASS



CLASS PremiumRide INHERITS Ride

    ATTRIBUTES
        ratePerKilometer : Double
        comfortSurcharge : Double


    CONSTRUCTOR PremiumRide(pickupLocation : String,dropOffLocation : String,distance : Double,route : Route,ratePerKilometer : Double,comfortSurcharge : Double)

        CALL SUPER(pickupLocation, dropOffLocation, distance, route)

        SET this.ratePerKilometer = ratePerKilometer

        SET this.comfortSurcharge = comfortSurcharge



    OVERRIDE METHOD calculateFare() : Double

        RETURN (distance * ratePerKilometer) + comfortSurcharge


END CLASS



CLASS SharedRide INHERITS StandardRide

    ATTRIBUTES
        numberOfPassengers : Integer


    CONSTRUCTOR SharedRide(pickupLocation : String,dropOffLocation : String,distance : Double,route : Route,ratePerKilometer : Double,numberOfPassengers : Integer)

        CALL SUPER(pickupLocation,dropOffLocation,distance,route,ratePerKilometer)


        SET this.numberOfPassengers = numberOfPassengers



    OVERRIDE METHOD calculateFare() : Double

        RETURN (distance * ratePerKilometer) / numberOfPassengers


END CLASS



CLASS Waypoint

    ATTRIBUTES
        location : String


    CONSTRUCTOR Waypoint(location : String)

        SET this.location = location


END CLASS



CLASS Route

    ATTRIBUTES
        waypoints : List(Waypoint)


    CONSTRUCTOR Route()

        CREATE empty waypoints list



    METHOD addWaypoint(waypoint : Waypoint)

        ADD waypoint TO waypoints



END CLASS
