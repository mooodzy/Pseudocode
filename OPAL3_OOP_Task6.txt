Task 303: Musical Instruments Store  


CLASS Instrument

    ATTRIBUTES
        brand : String
        price : Double

    CONSTRUCTOR Instrument(brand : String,price : Double)

        SET this.brand = brand
        SET this.price = price

    METHODS

        describePlayStyle() : String

            RETURN "Instrument playing style"

END CLASS



CLASS Guitar INHERITS Instrument

    ATTRIBUTES
        numberOfStrings : Integer


    CONSTRUCTOR Guitar(brand : String,price : Double,numberOfStrings : Integer)

        CALL SUPER(brand, price)

        SET this.numberOfStrings = numberOfStrings

    OVERRIDE METHOD describePlayStyle() : String

        RETURN "Strummed or plucked"

END CLASS



CLASS DrumKit INHERITS Instrument

    ATTRIBUTES
        numberOfDrums : Integer


    CONSTRUCTOR DrumKit(brand : String,
                        price : Double,
                        numberOfDrums : Integer)

        CALL SUPER(brand, price)

        SET this.numberOfDrums = numberOfDrums

    OVERRIDE METHOD describePlayStyle() : String

        RETURN "Struck"


END CLASS



CLASS Keyboard INHERITS Instrument

    ATTRIBUTES
        numberOfKeys : Integer


    CONSTRUCTOR Keyboard(brand : String,price : Double,numberOfKeys : Integer)

        CALL SUPER(brand, price)

        SET this.numberOfKeys = numberOfKeys


    OVERRIDE METHOD describePlayStyle() : String

        RETURN "Pressed"


END CLASS



CLASS Store

    ATTRIBUTES
        instruments : List<Instrument>


    CONSTRUCTOR Store()

        CREATE empty instruments list


    METHODS


        addInstrument(instrument : Instrument)

            ADD instrument TO instruments


        displayInstruments()

            SET i = 0

            WHILE i < size of instruments

                PRINT instruments[i].brand

                PRINT instruments[i].price

                PRINT instruments[i].describePlayStyle()

                SET i = i + 1

            END WHILE


END CLASS


START PROGRAM


DECLARE store1 : Store

SET store1 = NEW Store()



DECLARE guitar1 : Guitar

SET guitar1 = NEW Guitar("Yamaha",500.0, 6)


DECLARE drum1 : DrumKit

SET drum1 = NEW DrumKit("Pearl",800.0,5)


DECLARE keyboard1 : Keyboard

SET keyboard1 = NEW Keyboard("Casio",300.0,61)


CALL store1.addInstrument(guitar1)

CALL store1.addInstrument(drum1)

CALL store1.addInstrument(keyboard1)


CALL store1.displayInstruments()

END PROGRAM



Task 304: Fitness App Workouts



CLASS Workout

    ATTRIBUTES
        date : String
        duration : Double


    CONSTRUCTOR Workout(date : String,duration : Double)

        SET this.date = date
        SET this.duration = duration


    METHODS

        calculateCalories() : Double

            RETURN 0.0

END CLASS



CLASS RunningWorkout INHERITS Workout

    ATTRIBUTES
        distance : Double
        pace : Double


    CONSTRUCTOR RunningWorkout(date : String,duration : Double,distance : Double,pace : Double)

        CALL SUPER(date, duration)

        SET this.distance = distance
        SET this.pace = pace


    OVERRIDE METHOD calculateCalories() : Double

        RETURN distance * pace * 10

END CLASS



CLASS CyclingWorkout INHERITS Workout

    ATTRIBUTES
        distance : Double
        resistanceLevel : Integer


    CONSTRUCTOR CyclingWorkout(date : String,duration : Double,distance : Double,resistanceLevel : Integer)

        CALL SUPER(date, duration)
        SET this.distance = distance
        SET this.resistanceLevel = resistanceLevel


    OVERRIDE METHOD calculateCalories() : Double

        RETURN distance * resistanceLevel * 8

END CLASS



CLASS WeightTrainingWorkout INHERITS Workout

    ATTRIBUTES
        sets : Integer
        reps : Integer


    CONSTRUCTOR WeightTrainingWorkout(date : String,duration : Double,sets : Integer,reps : Integer)

        CALL SUPER(date, duration)
        SET this.sets = sets
        SET this.reps = reps


    OVERRIDE METHOD calculateCalories() : Double
        RETURN sets * reps * 2

END CLASS



CLASS FitnessLog

    ATTRIBUTES
        workouts : List(Workout)


    CONSTRUCTOR FitnessLog()

        CREATE empty workouts list


    METHODS

        addWorkout(workout : Workout)

            ADD workout TO workouts


        displayWorkouts()

            SET i = 0

            WHILE i < size of workouts

                PRINT workouts[i].date

                PRINT workouts[i].calculateCalories()

                SET i = i + 1

            END WHILE

END CLASS



START PROGRAM


DECLARE log1 : FitnessLog

SET log1 = NEW FitnessLog()



DECLARE run1 : RunningWorkout

SET run1 = NEW RunningWorkout("2026-07-08",45,8,6)

DECLARE cycle1 : CyclingWorkout

SET cycle1 = NEW CyclingWorkout("2026-07-09",60,20, 4)


DECLARE weight1 : WeightTrainingWorkout

SET weight1 = NEW WeightTrainingWorkout("2026-07-10", 50, 5,12)


CALL log1.addWorkout(run1)

CALL log1.addWorkout(cycle1)

CALL log1.addWorkout(weight1)


CALL log1.displayWorkouts()


END PROGRAM

Task 306: Utility Billing System


CLASS Bill

    ATTRIBUTES
        billingPeriod : String
        dueDate : String
        meterReading : MeterReading


    CONSTRUCTOR Bill(billingPeriod : String,dueDate : String,meterReading : MeterReading)

        SET this.billingPeriod = billingPeriod
        SET this.dueDate = dueDate
        SET this.meterReading = meterReading


    METHODS

        calculateAmountOwed() : Double

            RETURN 0.0

END CLASS



CLASS ElectricityBill INHERITS Bill

    ATTRIBUTES
        unitsConsumed : Double
        ratePerUnit : Double


    CONSTRUCTOR ElectricityBill(billingPeriod : String,dueDate : String,meterReading : MeterReading,unitsConsumed : Double,ratePerUnit : Double)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.unitsConsumed = unitsConsumed
        SET this.ratePerUnit = ratePerUnit


    OVERRIDE METHOD calculateAmountOwed() : Double

        RETURN unitsConsumed * ratePerUnit

END CLASS



CLASS WaterBill INHERITS Bill

    ATTRIBUTES
        litersConsumed : Double
        flatRate : Double
        standingCharge : Double


    CONSTRUCTOR WaterBill(billingPeriod : String,dueDate : String,meterReading : MeterReading,litersConsumed : Double,flatRate : Double,standingCharge : Double)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.litersConsumed = litersConsumed
        SET this.flatRate = flatRate
        SET this.standingCharge = standingCharge


    OVERRIDE METHOD calculateAmountOwed() : Double

        RETURN (litersConsumed * flatRate) + standingCharge

END CLASS



CLASS GasBill INHERITS Bill

    ATTRIBUTES
        unitsConsumed : Double
        ratePerUnit : Double
        seasonalSurcharge : Double


    CONSTRUCTOR GasBill(billingPeriod : String,dueDate : String, meterReading : MeterReading,unitsConsumed : Double,ratePerUnit : Double,seasonalSurcharge : Double)

        CALL SUPER(billingPeriod, dueDate, meterReading)

        SET this.unitsConsumed = unitsConsumed
        SET this.ratePerUnit = ratePerUnit
        SET this.seasonalSurcharge = seasonalSurcharge


    OVERRIDE METHOD calculateAmountOwed() : Double

        RETURN (unitsConsumed * ratePerUnit) + seasonalSurcharge

END CLASS



CLASS MeterReading

    ATTRIBUTES
        startReading : Double
        endReading : Double


    CONSTRUCTOR MeterReading(startReading : Double,endReading : Double)

        SET this.startReading = startReading
        SET this.endReading = endReading

END CLASS



CLASS CustomerAccount

    ATTRIBUTES
        bills : List<Bill>

    CONSTRUCTOR CustomerAccount()

        CREATE empty bills list


    METHODS

        addBill(bill : Bill)

            ADD bill TO bills


        displayBills()

            SET i = 0

            WHILE i < size of bills

                PRINT bills[i].billingPeriod

                PRINT bills[i].calculateAmountOwed()

            END WHILE

END CLASS

Task 307: Ride-Sharing Vehicle Tiers  


CLASS Ride

    ATTRIBUTES
        pickupLocation : String
        dropOffLocation : String
        route : Route


    CONSTRUCTOR Ride(pickupLocation : String,
                     dropOffLocation : String,
                     route : Route)

        SET this.pickupLocation = pickupLocation
        SET this.dropOffLocation = dropOffLocation
        SET this.route = route


    METHODS

        calculateFare() : Double

            RETURN 0.0

END CLASS



CLASS StandardRide INHERITS Ride

    ATTRIBUTES
        distance : Double
        ratePerKilometer : Double


    CONSTRUCTOR StandardRide(pickupLocation : String,dropOffLocation : String,route : Route,distance : Double,ratePerKilometer : Double)

        CALL SUPER(pickupLocation, dropOffLocation, route)

        SET this.distance = distance
        SET this.ratePerKilometer = ratePerKilometer


    OVERRIDE METHOD calculateFare() : Double

        RETURN distance * ratePerKilometer

END CLASS



CLASS PremiumRide INHERITS Ride

    ATTRIBUTES
        distance : Double
        ratePerKilometer : Double
        comfortSurcharge : Double


    CONSTRUCTOR PremiumRide(pickupLocation : String,dropOffLocation : String,route : Route,distance : Double,ratePerKilometer : Double ,comfortSurcharge : Double)

        CALL SUPER(pickupLocation, dropOffLocation, route)

        SET this.distance = distance
        SET this.ratePerKilometer = ratePerKilometer
        SET this.comfortSurcharge = comfortSurcharge


    OVERRIDE METHOD calculateFare() : Double

        RETURN (distance * ratePerKilometer) + comfortSurcharge

END CLASS



CLASS SharedRide INHERITS StandardRide

    ATTRIBUTES
        numberOfPassengers : Integer


    CONSTRUCTOR SharedRide(pickupLocation : String,dropOffLocation : String,route : Route,distance : Double,ratePerKilometer : Double,numberOfPassengers : Integer)

        CALL SUPER(pickupLocation,dropOffLocation,route,distance,ratePerKilometer)

        SET this.numberOfPassengers = numberOfPassengers


    OVERRIDE METHOD calculateFare() : Double

        RETURN (distance * ratePerKilometer) / numberOfPassengers

END CLASS



CLASS Waypoint

    ATTRIBUTES
        location : String


    CONSTRUCTOR Waypoint(location : String)

        SET this.location = location

END CLASS



CLASS Route

    ATTRIBUTES
        waypoints : List<Waypoint>


    CONSTRUCTOR Route()

        CREATE empty waypoints list


    METHODS

        addWaypoint(waypoint : Waypoint)

            ADD waypoint TO waypoints


END CLASS


START PROGRAM

DECLARE route1 : Route

SET route1 = NEW Route()


DECLARE point1 : Waypoint

SET point1 = NEW Waypoint("City Center")

CALL route1.addWaypoint(point1)



DECLARE ride1 : StandardRide
